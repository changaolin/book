template <int IDX, int MAX, typename... Args>
struct PRINT_TUPLE {
  static void print(ostream& os, const tuple<Args...>& t) {
    os << get<IDX>(t) << (IDX + 1 == MAX) ? "" : ",";
    PRINT_TUPLE<IDX+1, MAX, Args...>::print(os, t);
  }
}
